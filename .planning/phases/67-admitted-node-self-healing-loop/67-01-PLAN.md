# 67-01 — VlessProxyManager success_rate tracking + dead-node exclusion

**Phase:** 67 Admitted-Node Self-Healing Loop
**Requirements:** REL-15
**Depends on:** nothing

## Goal

Add per-host success/failure sample tracking to `VlessProxyManager`, expose `record_outcome` + `success_rate` public API, and exclude dead nodes (rate < 0.1 with ≥ 20 samples) from pool_snapshot + refresh_proxy_list.

### Public methods (add near `pool_snapshot`)

```python
def iter_admitted_hosts(self) -> list[str]:
    """Return the list of admitted VLESS host strings (not cooldown'd).

    Used by the re-probe loop to iterate what's currently in the pool
    without re-doing cooldown filtering in the caller.
    """
    now = time.time()
    with self._lock:
        nodes = list(self._pool.get("nodes", []))
        cooldowns = dict(self._cooldowns)
    active_cooldown = {
        h for h, e in cooldowns.items()
        if (now - float(e.get("blocked_at", 0.0))) < VKUSVILL_COOLDOWN_S
    }
    return [
        n["host"] for n in nodes
        if n.get("host") and n["host"] not in active_cooldown
    ]


def record_outcome(self, host: str | None, success: bool) -> None:
    """Record one production-traffic outcome against a host.

    ``host=None`` routes against the pool's current head-of-list (matches
    the heuristic in ``mark_current_node_blocked`` — xray's leastPing
    balancer selects the head of the pool most of the time, so this is
    close-enough for sliding-window sampling). Bounded to
    ``SUCCESS_RATE_WINDOW`` samples per host (FIFO).
    """
    if host is None:
        with self._lock:
            nodes = list(self._pool.get("nodes", []))
        if not nodes:
            return
        host = nodes[0].get("host")
        if not host:
            return

    now = time.time()
    with self._outcomes_lock:
        samples = self._outcomes.setdefault(host, [])
        samples.append(bool(success))
        if len(samples) > SUCCESS_RATE_WINDOW:
            del samples[: len(samples) - SUCCESS_RATE_WINDOW]
        if success:
            self._last_success_at[host] = now


def success_rate(self, host: str) -> float | None:
    """Return success rate over the window, or None if < 20 samples."""
    with self._outcomes_lock:
        samples = self._outcomes.get(host)
        if samples is None or len(samples) < SUCCESS_RATE_MIN_SAMPLES:
            return None
        return sum(samples) / len(samples)


def _is_node_dead(self, entry: dict) -> bool:
    """REL-15: exclude hosts whose success_rate is known-bad."""
    host = entry.get("host")
    if not host:
        return False
    rate = self.success_rate(host)
    return rate is not None and rate < SUCCESS_RATE_DEAD_THRESHOLD
```

## Tests — `tests/test_reprobe_loop.py` (tests 1-4 of 6)

```python
import threading
import pytest
from vless.manager import (
    VlessProxyManager,
    SUCCESS_RATE_WINDOW,
    SUCCESS_RATE_MIN_SAMPLES,
    SUCCESS_RATE_DEAD_THRESHOLD,
)


@pytest.fixture
def pm(tmp_path, monkeypatch):
    """Fresh manager with isolated pool/cooldown state."""
    pool_path = tmp_path / "vless_pool.json"
    pool_path.write_text('{"nodes": [{"host": "1.2.3.4", "port": 443}, {"host": "5.6.7.8", "port": 443}], "updated_at": "2026-05-12T00:00:00"}')
    cooldowns_path = tmp_path / "cooldowns.json"
    cooldowns_path.write_text("{}")
    events_path = tmp_path / "events.jsonl"
    xray_path = tmp_path / "active.json"
    monkeypatch.setattr("vless.manager._atomic_write_text", lambda path, content: None)
    m = VlessProxyManager(
        pool_path=pool_path,
        cooldowns_path=cooldowns_path,
        events_path=events_path,
        xray_config_path=xray_path,
        register_atexit=False,
        log_func=lambda _: None,
    )
    return m


def test_record_outcome_fifo_window(pm):
    """Exactly SUCCESS_RATE_WINDOW=100 samples max, oldest dropped first."""
    for i in range(150):
        pm.record_outcome("1.2.3.4", success=(i % 2 == 0))
    samples = pm._outcomes["1.2.3.4"]
    assert len(samples) == SUCCESS_RATE_WINDOW == 100
    # Last sample should be i=149 (odd -> False)
    assert samples[-1] is False
    # First sample should be i=50 (even -> True) — we dropped i=0..49
    assert samples[0] is True


def test_success_rate_unknown_below_20_samples(pm):
    for _ in range(SUCCESS_RATE_MIN_SAMPLES - 1):
        pm.record_outcome("1.2.3.4", success=True)
    assert pm.success_rate("1.2.3.4") is None


def test_success_rate_computed_above_20_samples(pm):
    for _ in range(5):
        pm.record_outcome("1.2.3.4", success=True)
    for _ in range(15):
        pm.record_outcome("1.2.3.4", success=False)
    rate = pm.success_rate("1.2.3.4")
    assert rate is not None
    assert abs(rate - 0.25) < 0.001


def test_dead_node_excluded_from_active_outbounds(pm):
    """Node with success_rate < 0.1 (and >= 20 samples) dropped from snapshot."""
    # 20 samples at 5% success -> rate=0.05 < 0.1 -> dead
    for i in range(20):
        pm.record_outcome("1.2.3.4", success=(i == 0))
    # Other node has no samples -> unknown -> alive
    snap = pm.pool_snapshot()
    assert snap["size"] == 2
    assert snap["dead_by_success_rate_count"] == 1
    assert snap["active_outbounds"] == 1  # only 5.6.7.8 is alive
```

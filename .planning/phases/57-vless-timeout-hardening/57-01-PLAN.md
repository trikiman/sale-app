def test_build_xray_config_has_policy_block(reality_node):
    config = build_xray_config([reality_node])
    assert "policy" in config
    level0 = config["policy"]["levels"]["0"]
    assert level0["connIdle"] == 30, "connIdle must be 30s, not xray default 300s"
    assert level0["handshake"] == 8, "handshake must be 8s to fit VLESS+Reality (3-5s observed)"
    assert level0["bufferSize"] == 4096
    assert level0["statsUserUplink"] is False
    assert level0["statsUserDownlink"] is False


def test_build_xray_config_has_observatory(reality_node):
    config = build_xray_config([reality_node])
    assert "observatory" in config
    obs = config["observatory"]
    assert obs["subjectSelector"] == ["node-"]
    assert obs["probeUrl"] == "https://www.google.com/generate_204"
    assert obs["probeInterval"] == "5m"


def test_build_xray_config_balancer_uses_least_ping(reality_node, another_reality_node):
    config = build_xray_config([reality_node, another_reality_node])
    balancer = config["routing"]["balancers"][0]
    assert balancer["strategy"] == {"type": "leastPing"}
    # selector still references every outbound
    assert set(balancer["selector"]) == {"node-0", "node-1"}


def test_build_xray_config_is_json_serializable_with_new_sections(reality_node):
    import json
    config = build_xray_config([reality_node])
    # Must round-trip without raising (xray reads this as JSON)
    serialized = json.dumps(config, indent=2)
    roundtripped = json.loads(serialized)
    assert roundtripped["policy"]["levels"]["0"]["connIdle"] == 30
    assert roundtripped["observatory"]["probeInterval"] == "5m"

## Commit Message

```
fix(vless): add xray policy + observatory + leastPing balancer (phase 57-01)

Phase 56 (v1.15) shipped an xray config without a policy block, which
left connIdle at xray's default of 300 seconds. When a VLESS upstream
node slowed down mid-stream, xray held the dead TCP connection open for
5 full minutes, causing the "middle of connection timeout" the user
reported on 2026-04-23.

Phase 56 also deferred leastPing in favor of random balancer strategy
per plan D-08. "Deferred" was interpreted as "don't wire observatory at
all," leaving dead outbounds in rotation until the 24h pool refresh.
Every Nth request randomly hit a dead outbound and hung for 5 minutes
(compound of the policy gap above).

Fix:
- vless/config_gen.py: add explicit policy.levels["0"] with connIdle=30,
  handshake=8, bufferSize=4096; add observatory probing every outbound
  whose tag starts with "node-" via generate_204 every 5 minutes; change
  balancer strategy from random to leastPing so dead/slow outbounds are
  automatically deprioritized based on observatory data.
- tests/test_vless_config_gen.py: 4 new tests asserting the policy +
  observatory + leastPing blocks are present and JSON-serializable.

Primary diagnosis in .planning/phases/56-vless-proxy-migration/
INSPECTION-2026-04-23.md (sections R1, R2).

Tests: 167/2 → 171/2 passing. Ruff clean.
```

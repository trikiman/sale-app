```python
def test_fresh_breaker_starts_closed():
    b = sch.BreakerState()

def test_closed_trips_to_open_after_threshold_all_fail_cycles():
    b = sch.BreakerState()
    for _ in range(sch.BREAKER_TRIP_THRESHOLD):
        b.record_all_failed()
    assert b.state == "open"
    assert b.cooldown_until_ts > 0


def test_open_transitions_to_half_open_after_cooldown(monkeypatch):
    b = sch.BreakerState()
    for _ in range(sch.BREAKER_TRIP_THRESHOLD):
        b.record_all_failed()
    # advance wall clock past cooldown
    import time as _t
    monkeypatch.setattr(sch.time, "time", lambda: b.cooldown_until_ts + 1)
    b.tick()
    assert b.state == "half_open"


def test_half_open_success_resets_to_closed(monkeypatch):
    b = sch.BreakerState()
    for _ in range(sch.BREAKER_TRIP_THRESHOLD):
        b.record_all_failed()
    monkeypatch.setattr(sch.time, "time", lambda: b.cooldown_until_ts + 1)
    b.tick()
    assert b.state == "half_open"
    b.record_any_success()
    assert b.state == "closed"
    assert b.fails == 0
    assert b.cooldown_s == sch.BREAKER_BASE_COOLDOWN_S


def test_half_open_fail_trips_back_to_open_with_doubled_cooldown(monkeypatch):
    b = sch.BreakerState()
    for _ in range(sch.BREAKER_TRIP_THRESHOLD):
        b.record_all_failed()
    initial_cooldown = b.cooldown_s
    monkeypatch.setattr(sch.time, "time", lambda: b.cooldown_until_ts + 1)
    b.tick()
    b.record_all_failed()
    assert b.state == "open"
    assert b.cooldown_s == initial_cooldown * 2

    b = sch.BreakerState(cooldown_s=sch.BREAKER_MAX_COOLDOWN_S)

    monkeypatch.setattr(sch.time, "time", lambda: now)
    b.state = "half_open"
    b.record_all_failed()
    assert b.cooldown_s == sch.BREAKER_MAX_COOLDOWN_S
```

## Probe URL regression tests

```python
"""Phase 60 tests — xray observatory probeURL regression guard (REL-06)."""
import pytest

from vless.config_gen import build_xray_config, VlessNode


def _fake_node():
    # Minimal node constructor; adjust to whatever the real class wants.
    # Use a placeholder — the real VlessNode is probably a dataclass.
    return VlessNode(
        host="127.0.0.1", port=443, uuid="00000000-0000-0000-0000-000000000000",
        flow="", sni="example.com", fp="chrome", public_key="x", short_id="00",
    )


def test_probe_url_targets_vkusvill_not_google():
    cfg = build_xray_config([_fake_node()])
    probe = cfg["observatory"]["probeURL"]
    assert "vkusvill.ru" in probe, (
        f"probeURL regressed to '{probe}'. "
        "Must target vkusvill.ru/favicon.ico per REL-06."
    )
    assert "google.com" not in probe, (
        f"probeURL regressed to Google: '{probe}'. "
        "This was the silent-killer root cause: leastPing ranked by Google ping "
        "instead of VkusVill reachability."
    )


def test_probe_interval_is_at_most_sixty_seconds():
    cfg = build_xray_config([_fake_node()])
    interval = cfg["observatory"]["probeInterval"]
    # Accept 60s / 30s / 15s / 10s; reject "5m" and larger.
    assert interval.endswith("s"), f"probeInterval '{interval}' must be seconds"
    value = int(interval.rstrip("s"))
    assert value <= 60, f"probeInterval {interval} is > 60s (slow-detection regression)"


def test_balancer_uses_leastping_strategy():
    cfg = build_xray_config([_fake_node()])
    balancers = cfg["routing"]["balancers"]
    assert balancers, "at least one balancer must exist"
    assert balancers[0]["strategy"]["type"] == "leastPing", (
        "probeURL fix is only meaningful when leastPing is the strategy; "
        "do not regress to random or roundRobin without revisiting REL-06."
    )
```

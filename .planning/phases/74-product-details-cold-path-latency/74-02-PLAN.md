### `backend/detail_events.py` (new)

Mirrors `backend/cart_events.py` (v1.20 OBS-05) pattern.

```python
"""Bounded JSONL ledger for product-detail endpoint timing/outcomes.

Schema (one line per /api/product/{id}/details call):
    {"ts": 1747123456.789, "product_id": "33215",
     "duration_ms": 1823, "cached": false,
     "retry_count": 1, "outcome": "ok"}

Outcomes:
    "ok"        — 200 response, >500 bytes, parsed successfully
    "fallback"  — fetched but html too short / unreadable
    "failed"    — all retries exhausted with exception/non-200
    "cached"    — cache hit, no network call (retry_count=0)

Bounded via MAX_LINES prune; same approach as backend/cart_events.py.
Callers must never crash the endpoint on ledger I/O errors.
"""

from __future__ import annotations

import json
import logging
import os
import time
from typing import Literal

from backend import BASE_PROJECT_DIR  # or equivalent data-dir anchor

Outcome = Literal["ok", "fallback", "failed", "cached"]

LEDGER_PATH = os.path.join(BASE_PROJECT_DIR, "data", "detail_events.jsonl")
MAX_LINES = 5000          # ~0.5 MB at ~100 bytes/line
PRUNE_KEEP = 4000         # after prune, keep the most recent N

_logger = logging.getLogger(__name__)


def append_event(
    *,
    product_id: str,
    duration_ms: int,
    cached: bool,
    retry_count: int,
    outcome: Outcome,
) -> None:
    """Append one event. Never raises — ledger write must not crash caller."""
    entry = {
        "ts": round(time.time(), 3),
        "product_id": str(product_id),
        "duration_ms": int(duration_ms),
        "cached": bool(cached),
        "retry_count": int(retry_count),
        "outcome": outcome,
    }
    try:
        os.makedirs(os.path.dirname(LEDGER_PATH), exist_ok=True)
        with open(LEDGER_PATH, "a", encoding="utf-8") as fh:
            fh.write(json.dumps(entry, separators=(",", ":")) + "\n")
        _maybe_prune()
    except Exception:  # noqa: BLE001 - ledger must never crash the endpoint
        _logger.warning("detail_events append failed", exc_info=True)


def _maybe_prune() -> None:
    """Prune when file exceeds MAX_LINES. Keeps most recent PRUNE_KEEP lines."""
    try:
        if not os.path.exists(LEDGER_PATH):
            return
        with open(LEDGER_PATH, "r", encoding="utf-8") as fh:
            lines = fh.readlines()
        if len(lines) <= MAX_LINES:
            return
        keep = lines[-PRUNE_KEEP:]
        tmp = LEDGER_PATH + ".tmp"
        with open(tmp, "w", encoding="utf-8") as fh:
            fh.writelines(keep)
        os.replace(tmp, LEDGER_PATH)
    except Exception:  # noqa: BLE001
        _logger.warning("detail_events prune failed", exc_info=True)


def read_recent(limit: int = 100) -> list[dict]:
    """Tail the last N events. Used by admin/tests."""
    if not os.path.exists(LEDGER_PATH):
        return []
    try:
        with open(LEDGER_PATH, "r", encoding="utf-8") as fh:
            lines = fh.readlines()
        out = []
        for line in lines[-limit:]:
            try:
                out.append(json.loads(line))
            except json.JSONDecodeError:
                continue
        return out
    except Exception:  # noqa: BLE001
        return []
```

### `backend/test_product_details_latency.py` (new — force-add, `backend/test_*.py` is gitignored)

Mocks httpx with `respx` (already in requirements via other tests). Covers three cases:

```python
"""Phase 74 — product_details latency ledger tests.

Verifies detail_events.jsonl records correct outcome + retry_count for:
- cache hit (outcome="cached", retry_count=0)
- happy-path fetch (outcome="ok", retry_count=1 or more if first attempt non-200)
- all retries fail (outcome="failed", retry_count=3)
- fallback when HTML is too short (outcome="fallback")
"""

from __future__ import annotations

import json
import os
from pathlib import Path

import httpx
import pytest
from fastapi.testclient import TestClient

# Force ledger to a tmp path before importing modules
@pytest.fixture
def tmp_ledger(tmp_path, monkeypatch):
    ledger = tmp_path / "detail_events.jsonl"
    monkeypatch.setenv("SALEAPP_DETAIL_EVENTS_PATH", str(ledger))
    # Reload module so LEDGER_PATH picks up the env var override
    from backend import detail_events
    monkeypatch.setattr(detail_events, "LEDGER_PATH", str(ledger))
    yield ledger


def _read_events(path: Path) -> list[dict]:
    if not path.exists():
        return []
    with open(path, encoding="utf-8") as fh:
        return [json.loads(line) for line in fh if line.strip()]


def test_cached_path_emits_cached_outcome(tmp_ledger, monkeypatch):
    from backend import detail_service, main
    monkeypatch.setattr(detail_service, "read_cache", lambda pid: {"id": pid, "cached": True})
    monkeypatch.setattr(main, "_load_product_record", lambda pid: {"id": pid, "url": "https://vkusvill.ru/goods/x.html"})

    with TestClient(main.app) as client:
        resp = client.get("/api/product/33215/details")

    assert resp.status_code == 200
    events = _read_events(tmp_ledger)
    assert len(events) == 1
    assert events[0]["outcome"] == "cached"
    assert events[0]["cached"] is True
    assert events[0]["retry_count"] == 0
    assert events[0]["product_id"] == "33215"


def test_happy_fetch_emits_ok_outcome(tmp_ledger, monkeypatch, respx_mock):
    from backend import detail_service, main
    monkeypatch.setattr(detail_service, "read_cache", lambda pid: None)
    monkeypatch.setattr(detail_service, "write_cache", lambda pid, result: None)
    monkeypatch.setattr(main, "_load_product_record", lambda pid: {
        "id": pid, "url": "https://vkusvill.ru/goods/33215.html"
    })

    respx_mock.get("https://vkusvill.ru/goods/33215.html").mock(
        return_value=httpx.Response(200, text="<html>" + "x" * 600 + "</html>")
    )

    with TestClient(main.app) as client:
        resp = client.get("/api/product/33215/details")

    assert resp.status_code == 200
    events = _read_events(tmp_ledger)
    assert len(events) == 1
    assert events[0]["outcome"] == "ok"
    assert events[0]["cached"] is False
    assert events[0]["retry_count"] == 1


def test_all_retries_fail_emits_failed_outcome(tmp_ledger, monkeypatch, respx_mock):
    from backend import detail_service, main
    monkeypatch.setattr(detail_service, "read_cache", lambda pid: None)
    monkeypatch.setattr(main, "_load_product_record", lambda pid: {
        "id": pid, "url": "https://vkusvill.ru/goods/33215.html"
    })

    respx_mock.get("https://vkusvill.ru/goods/33215.html").mock(
        side_effect=httpx.ConnectError("bridge down")
    )

    with TestClient(main.app) as client:
        resp = client.get("/api/product/33215/details")

    assert resp.status_code == 200  # fallback still returns 200
    body = resp.json()
    assert body.get("source_unavailable") is True
    events = _read_events(tmp_ledger)
    assert len(events) == 1
    assert events[0]["outcome"] == "failed"
    assert events[0]["retry_count"] == 3


def test_short_html_emits_fallback_outcome(tmp_ledger, monkeypatch, respx_mock):
    from backend import detail_service, main
    monkeypatch.setattr(detail_service, "read_cache", lambda pid: None)
    monkeypatch.setattr(main, "_load_product_record", lambda pid: {
        "id": pid, "url": "https://vkusvill.ru/goods/33215.html"
    })

    respx_mock.get("https://vkusvill.ru/goods/33215.html").mock(
        return_value=httpx.Response(200, text="<html>tiny</html>")
    )

    with TestClient(main.app) as client:
        resp = client.get("/api/product/33215/details")

    assert resp.status_code == 200
    events = _read_events(tmp_ledger)
    assert len(events) == 1
    assert events[0]["outcome"] == "fallback"
```

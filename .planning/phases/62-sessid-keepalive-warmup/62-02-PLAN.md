In `cart_items_endpoint(user_id, request)` (body starts at line 3739), insert right after `_validate_user_header(request, user_id)` (line 3740) and before `cookies_path = _resolve_cart_cookies_path(user_id)` (line 3741):

```python
    # v1.20 PERF-04: on-app-open warmup nudge. Non-blocking; the daemon
    # checks anti-spam (<=1/user/15 min) + CART_ADD_ACTIVE flag.
    try:
        NUDGE_QUEUE.put_nowait(_keepalive_hash_user_id(str(user_id)))
    except Exception:
        logger.debug("keepalive nudge (cart_items) dropped", exc_info=False)
```

**Why `put_nowait`:** the queue is sized 256 (ample for 5 users × bursts) and we never want to block the API response. If the queue is full (catastrophic backlog), dropping the nudge is strictly better than adding latency — the 20-min cycle will still pick it up.

**Why `_hash_user_id` and not the raw id:** downstream JSONL records the hash; keeping the nudge queue in the same representation lets the daemon route directly without a second hash step.

### Step 2 — Edit `cart/vkusvill_api.py::add()`: wire CART_ADD_ACTIVE flag

At the top of `add()` — right after `t_start = time.monotonic()` (line 411) and BEFORE `self._ensure_session()` (line 412):

```python
        # v1.20 D4: flag cart-add as active so the keepalive daemon can
        # cancel any in-flight warmup for this user. Cleared in the
        # finally at the bottom of this method. Best-effort; lazy import
        # avoids a circular with keepalive.warmup.
        _keepalive_hash: str | None = None
        _CART_ADD_ACTIVE = None
        try:
            from keepalive.warmup import CART_ADD_ACTIVE as _CART_ADD_ACTIVE, _hash_user_id as _kp_hash
            ident = str(self.user_id) if self.user_id else ""
            if not ident:
                # user_id may not be loaded yet (filled during _ensure_session).
                # Fall back to cookies file basename which is stable per user.
                ident = os.path.basename(self.cookies_path or "")
            if ident:
                _keepalive_hash = _kp_hash(ident)
                _CART_ADD_ACTIVE[_keepalive_hash] = time.monotonic()
        except Exception:
            _keepalive_hash = None
```

Wrap the existing body from here onward in a `try/finally`. The shape is:

```python
        try:
            self._ensure_session()
            # ... all existing body of add() unchanged: the `for _ in range(quantity)`
            # loop, result parsing, return result.
            ...
        finally:
            if _keepalive_hash is not None and _CART_ADD_ACTIVE is not None:
                try:
                    _CART_ADD_ACTIVE.pop(_keepalive_hash, None)
                except Exception:
                    pass
```

**Notes for the executor:**
- `self.user_id` is populated during `_ensure_session()` (line 412), so at the top of `add()` it may still be `None`. That's fine: we hash the cookies file basename as a stable fallback identifier. This MUST match whatever the warmup side hashes (see 62-01 `_collect_linked_users`: it hashes the telegram_id/phone that maps to the cookies path). If the two hash inputs diverge, the flag races are harmless but the cancellation never fires. Document this as an accepted edge case: D4 is best-effort by CONTEXT design.
- The import MUST be lazy (inside the function body, not top-of-file) to avoid a circular import at module load: `cart.vkusvill_api` is imported by `backend/main.py`, which also imports `keepalive.warmup` → which (lazily) reaches into `cart.vkusvill_api` for the warmup HTTP call.

from __future__ import annotations

import json
import os
import threading
import time
from datetime import datetime, timezone
from pathlib import Path

import pytest

from keepalive import warmup


@pytest.fixture(autouse=True)
def _reset_state(tmp_path, monkeypatch):
    """Clear module state between tests; redirect JSONL into tmp_path."""
    warmup.reset_state_for_tests()
    monkeypatch.setattr(warmup, "WARMUP_JSONL_PATH", str(tmp_path / "warmup_events.jsonl"))
    monkeypatch.setattr(warmup, "BOOT_GRACE_S", 0.0)
    yield
    warmup.reset_state_for_tests()


@pytest.fixture
def three_linked_users(tmp_path, monkeypatch):
    """Fake 3 users with cookies.json files; monkeypatch _collect_linked_users."""
    users = []
    for tg_id in ("111", "222", "333"):
        p = tmp_path / f"{tg_id}.json"
        p.write_text(json.dumps({"cookies": [], "sessid": "old", "user_id": tg_id, "sessid_ts": 1}))
        users.append((warmup._hash_user_id(tg_id), str(p)))
    monkeypatch.setattr(warmup, "_collect_linked_users", lambda: list(users))
    return users


def _read_jsonl() -> list[dict]:
    path = warmup.WARMUP_JSONL_PATH
    if not os.path.exists(path):
        return []
    return [json.loads(ln) for ln in Path(path).read_text().splitlines() if ln.strip()]


def test_jsonl_schema(three_linked_users, monkeypatch):
    monkeypatch.setattr(warmup, "_pool_is_healthy", lambda: True)
    monkeypatch.setattr(
        warmup, "_warmup_single_user",
        lambda h, p, t, e: warmup._emit_event({
            "timestamp_iso": datetime.now(timezone.utc).isoformat(),
            "user_id_hash": h, "trigger": t, "outcome": "ok",
            "latency_ms": 100, "sessid_changed": True,
        }),
    )
    warmup._run_cycle(threading.Event())
    events = _read_jsonl()
    assert len(events) == 3
    required = {"timestamp_iso", "user_id_hash", "trigger", "outcome", "latency_ms", "sessid_changed"}
    for e in events:
        assert required <= set(e.keys())
        assert isinstance(e["latency_ms"], int)
        assert isinstance(e["sessid_changed"], bool)
        assert e["trigger"] in {"scheduler", "app_open"}


def test_anti_spam_ceiling(three_linked_users, monkeypatch):
    monkeypatch.setattr(warmup, "_pool_is_healthy", lambda: True)
    count = {"n": 0}

    def fake_warm(hash_id, path, trigger, stop_event):
        count["n"] += 1
        with warmup._STATE_LOCK:
            warmup._LAST_WARMUP_AT[hash_id] = time.monotonic()
        warmup._emit_event({"timestamp_iso": "t", "user_id_hash": hash_id,
                            "trigger": trigger, "outcome": "ok",
                            "latency_ms": 100, "sessid_changed": True})
    monkeypatch.setattr(warmup, "_warmup_single_user", fake_warm)

    warmup._run_cycle(threading.Event())
    assert count["n"] == 3
    # 10 min later, still inside 15-min window
    t_later = time.monotonic() + 10 * 60
    monkeypatch.setattr(time, "monotonic", lambda: t_later)
    warmup._run_cycle(threading.Event())
    assert count["n"] == 3  # all 3 skipped
    events = _read_jsonl()
    outcomes = [e["outcome"] for e in events]
    assert outcomes.count("ok") == 3
    assert outcomes.count("skipped_recent") == 3


def test_pool_unhealthy_gate(three_linked_users, monkeypatch):
    monkeypatch.setattr(warmup, "_pool_is_healthy", lambda: False)
    called = {"n": 0}
    monkeypatch.setattr(warmup, "_warmup_single_user",
                        lambda *a, **k: called.__setitem__("n", called["n"] + 1))
    warmup._run_cycle(threading.Event())
    assert called["n"] == 0
    events = _read_jsonl()
    assert [e["outcome"] for e in events] == ["skipped_unhealthy"] * 3


def test_breaker_open_gate(three_linked_users, monkeypatch):
    # Unit-level stand-in for the full gate — if _pool_is_healthy returns
    # False for any reason (breaker open counts), cycle is skipped.
    monkeypatch.setattr(warmup, "_pool_is_healthy", lambda: False)
    warmup._run_cycle(threading.Event())
    assert all(e["outcome"] == "skipped_unhealthy" for e in _read_jsonl())


def test_cart_add_active_cancellation(three_linked_users, monkeypatch):

    http_called = {"n": 0}

        "cart.vkusvill_api.VkusVillCart._refresh_stale_session",
        lambda self: http_called.__setitem__("n", http_called["n"] + 1),

        warmup.CART_ADD_ACTIVE[three_linked_users[0][0]] = time.monotonic()

    cancelled = [e for e in events if e["outcome"] == "cancelled_by_cart_add"]
    assert len(cancelled) == 1
    assert cancelled[0]["user_id_hash"] == three_linked_users[0][0]
    # http_called["n"] should be 2 (the OTHER two users warmed normally),
    # not 3 — the flagged user skipped the HTTP call entirely.
    assert http_called["n"] == 2


def test_stop_event_clean_shutdown(three_linked_users, monkeypatch):
    monkeypatch.setattr(warmup, "_pool_is_healthy", lambda: True)
    monkeypatch.setattr(warmup, "_warmup_single_user",
                        lambda *a, **k: time.sleep(0.2))
    stop = threading.Event()
    t = threading.Thread(target=warmup.start_warmup_loop, args=(stop,), daemon=True)
    t.start()
    time.sleep(0.3)
    stop.set()
    t.join(timeout=5.0)
    assert not t.is_alive()


def test_jsonl_rotation_at_ten_mb():
    p = Path(warmup.WARMUP_JSONL_PATH)
    # Seed just over 10 MB
    p.write_bytes(b'{"x":1}\n' * (warmup.JSONL_MAX_BYTES // 8 + 1000))

    assert rotated.exists()
    assert p.exists()
    assert len(p.read_text().strip().splitlines()) == 1

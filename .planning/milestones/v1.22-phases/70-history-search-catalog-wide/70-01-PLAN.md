### `backend/main.py` — new helper near `_load_admitted_host_set`

```python
def _load_current_sale_types() -> dict[str, str]:
    """Return {product_id_str: 'green' | 'red' | 'yellow'} from today's proposals.json.

    v1.22 Phase 70 UX-BUG-01. Source of truth for "is this product on sale
    right now and what color?" — distinct from sale_sessions (which is
    sale-continuity truth, can lag by up to one scraper cycle).

    Empty dict on missing/malformed PROPOSALS_PATH — search results then
    fall back to the historical last_sale_type, same defensive pattern as
    _load_admitted_host_set.
    """
    try:
        with open(PROPOSALS_PATH, "r", encoding="utf-8") as fh:
            data = json.load(fh)
    except (OSError, json.JSONDecodeError):
        return {}
    out: dict[str, str] = {}
    for product in data.get("products", []) or []:
        pid = product.get("id")
        ptype = product.get("type")
        if pid is None or not isinstance(ptype, str):
            continue
        if ptype not in ("green", "red", "yellow"):
            continue
        out[str(pid)] = ptype
    return out
```

"""v1.22 Phase 70 UX-BUG-01 — history search catalog-wide + currentSaleType."""
from __future__ import annotations

import json
import os
import pathlib
import pytest

import backend.main as bm


def test_load_current_sale_types_empty_on_missing_file(tmp_path, monkeypatch):
    """Missing proposals.json → empty dict, not exception."""
    monkeypatch.setattr(bm, "PROPOSALS_PATH", str(tmp_path / "nope.json"))
    assert bm._load_current_sale_types() == {}


def test_load_current_sale_types_empty_on_malformed(tmp_path, monkeypatch):
    """Malformed JSON → empty dict."""
    p = tmp_path / "proposals.json"
    p.write_text("{not json", encoding="utf-8")
    monkeypatch.setattr(bm, "PROPOSALS_PATH", str(p))
    assert bm._load_current_sale_types() == {}


def test_load_current_sale_types_maps_id_to_type(tmp_path, monkeypatch):
    """Valid proposals.json → {id -> type} map for green/red/yellow only."""
    p = tmp_path / "proposals.json"
    p.write_text(
        json.dumps(
            {
                "products": [
                    {"id": 1001, "type": "green"},
                    {"id": 1002, "type": "red"},
                    {"id": 1003, "type": "yellow"},
                    {"id": 1004, "type": "other"},       # should be skipped
                    {"id": None, "type": "green"},        # should be skipped
                    {"id": 1005, "type": None},           # should be skipped
                ]
            }
        ),
        encoding="utf-8",
    )
    monkeypatch.setattr(bm, "PROPOSALS_PATH", str(p))
    result = bm._load_current_sale_types()
    assert result == {"1001": "green", "1002": "red", "1003": "yellow"}


def test_history_search_exposes_currentSaleType_for_live_product(
    tmp_path, monkeypatch
):
    """Fixture: product 2001 has NO sale session, IS in today's proposals as
    green. History search should return it with currentSaleType='green' AND
    is_currently_on_sale=True."""
    import sqlite3
    # Set up a throwaway sqlite DB with product_catalog + sale_sessions.
    dbp = tmp_path / "test.db"
    conn = sqlite3.connect(str(dbp))
    c = conn.cursor()
    c.execute("""CREATE TABLE product_catalog (
        product_id INTEGER PRIMARY KEY, name TEXT, category TEXT,
        group_name TEXT, subgroup TEXT, image_url TEXT, last_known_price REAL,
        total_sale_count INTEGER, last_sale_at TEXT, last_sale_type TEXT,
        avg_discount_pct REAL, max_discount_pct REAL,
        usual_sale_time TEXT, avg_catch_window_min REAL
    )""")
    c.execute("""CREATE TABLE sale_sessions (
        id INTEGER PRIMARY KEY AUTOINCREMENT, product_id INTEGER,
        is_active INTEGER, old_price REAL
    )""")
    # Product 2001: no history, no active session — in proposals as green.
    c.execute(
        "INSERT INTO product_catalog VALUES (2001, 'цезарь салат', 'cat',"
        " 'Готовая еда', NULL, NULL, 299, 0, NULL, NULL, 0, 0, NULL, 0)"
    )
    # Product 2002: has history (last_sale_type=yellow), not in proposals.
    c.execute(
        "INSERT INTO product_catalog VALUES (2002, 'цезарь ролл', 'cat',"
        " 'Готовая еда', NULL, NULL, 199, 5, '2026-05-01', 'yellow',"
        " 15, 25, NULL, 30)"
    )
    conn.commit()
    conn.close()

    # Seed proposals.json with product 2001 as green.
    proposals = tmp_path / "proposals.json"
    proposals.write_text(
        json.dumps({"products": [{"id": 2001, "type": "green"}]}),
        encoding="utf-8",
    )

    # Patch backend module to use this test DB + proposals.
    monkeypatch.setattr(bm, "PROPOSALS_PATH", str(proposals))
    # The endpoint uses db.db_path — monkeypatch the Database instance.
    monkeypatch.setattr(bm.db, "db_path", str(dbp))

    # Call the handler directly (bypassing FastAPI TestClient for simplicity).
    # We invoke the underlying function.
    response = bm.history_get_products(
        page=1, per_page=50, search="цезарь",
        category=None, group=None, subgroup=None,
        filter=None, sort="last_seen",
        x_telegram_user_id=None,
    )
    assert response["total"] >= 2
    by_id = {p["id"]: p for p in response["products"]}
    # Product 2001 is live → currentSaleType=green AND is_currently_on_sale=True
    assert by_id[2001]["currentSaleType"] == "green"
    assert by_id[2001]["is_currently_on_sale"] is True
    # Product 2002 is history-only → currentSaleType=None
    assert by_id[2002]["currentSaleType"] is None
    assert by_id[2002]["last_sale_type"] == "yellow"

## Goal

Add a new helper `_compute_xray_drift_block(pool: dict)` to `backend/main.py` that computes the `xray_drift` block + degraded/unhealthy reason + critical flag. Add in-memory first-seen tracking keyed by the drifted-set frozenset. Add threshold constants. Do NOT yet wire into `_build_reliability_snapshot` — that's 69-02. This commit is purely additive.

### `backend/main.py` — new helper: extract running xray hosts

Near `_check_xray_listening`:

```python
def _extract_running_xray_hosts_for_health() -> set[str] | None:
    """Mirror of VlessProxyManager._extract_running_hosts for /api/health/deep.

    Reads bin/xray/configs/active.json and returns the set of outbound
    host addresses. Returns None (distinct from empty set) on missing or
    malformed config — the drift block is then absent rather than reporting
    spurious drift against "no hosts".
    """
    from pathlib import Path
    cfg_path = Path(__file__).resolve().parent.parent / "bin" / "xray" / "configs" / "active.json"
    try:
        with cfg_path.open("r", encoding="utf-8") as fh:
            config = json.load(fh)
    except (OSError, json.JSONDecodeError):
        return None
    hosts: set[str] = set()
    for outbound in config.get("outbounds", []) or []:
        settings = outbound.get("settings") or {}
        for vnext_entry in settings.get("vnext") or []:
            addr = vnext_entry.get("address")
            if isinstance(addr, str) and addr:
                hosts.add(addr)
    return hosts
```

### `backend/main.py` — `_compute_xray_drift_block`

After `_compute_cart_add_block`:

```python
def _compute_xray_drift_block(
    pool: dict,
    cycle_age: float | None,
) -> tuple[dict | None, str | None, bool]:
    """Compute the xray_drift block for /api/health/deep.

    Returns ``(block, reason, is_critical)``.

    * ``block`` is ``None`` when pool snapshot is unavailable OR when the
      running xray config can't be read — same no-ledger fallback as
      ``_compute_cart_add_block`` (v1.20 OBS-04).
    * ``reason`` is non-None only when drift has persisted for more than
      ``_DEEP_DRIFT_DEGRADED_S`` seconds (OBS-06 degraded threshold).
    * ``is_critical`` is True when drift is present AND cycle age exceeds
      ``_DEEP_DRIFT_UNHEALTHY_CYCLE_AGE_S`` — promotes severity to
      unhealthy in ``_build_reliability_snapshot``.

    First-seen tracking uses ``frozenset(drifted_hosts)`` as key so that
    "drift resolved then a different drift appeared" correctly resets the
    clock (see 69-CONTEXT.md §D4).
    """
    if not pool or not pool.get("available"):
        return None, None, False

    active_set = _extract_running_xray_hosts_for_health()
    if active_set is None:
        return None, None, False

    admitted_count = int(pool.get("size", 0))
    # We don't have the explicit admitted host list in the snapshot
    # (pool_snapshot is a count-only summary). Get the hosts from the
    # same pool JSON path.
    admitted_set = _load_admitted_host_set()
    if admitted_set is None:
        return None, None, False

    drifted = admitted_set.symmetric_difference(active_set)
    drift_count = len(drifted)
    drifted_hosts_sorted = sorted(drifted)

    # First-seen bookkeeping
    now_mono = _time.monotonic()
    drifted_key = frozenset(drifted)

    # Prune entries whose set no longer matches (drift resolved or shifted).
    for stale_key in list(_DRIFT_FIRST_SEEN.keys()):
        if stale_key != drifted_key:
            _DRIFT_FIRST_SEEN.pop(stale_key, None)

    first_seen_iso: str | None = None
    if drift_count > 0:
        if drifted_key in _DRIFT_FIRST_SEEN:
            first_seen_mono, first_seen_iso = _DRIFT_FIRST_SEEN[drifted_key]
        else:
            first_seen_iso = datetime.now().isoformat(timespec="seconds")
            _DRIFT_FIRST_SEEN[drifted_key] = (now_mono, first_seen_iso)
            first_seen_mono = now_mono
    else:
        first_seen_mono = 0.0

    block = {
        "admitted_hosts": len(admitted_set),
        "active_outbounds": len(active_set),
        "drift_count": drift_count,
        "drifted_hosts": drifted_hosts_sorted,
        "first_seen_at": first_seen_iso,
    }

    reason: str | None = None
    is_critical = False
    if drift_count > 0:
        age_s = now_mono - first_seen_mono
        if age_s >= _DEEP_DRIFT_DEGRADED_S:
            reason = f"xray_stale_config:{drift_count}_nodes_drifted"
            if cycle_age is not None and cycle_age > _DEEP_DRIFT_UNHEALTHY_CYCLE_AGE_S:
                is_critical = True

    return block, reason, is_critical


def _load_admitted_host_set() -> set[str] | None:
    """Load the set of admitted host strings from data/vless_pool.json.

    Returns None on missing/malformed file.
    """
    from pathlib import Path
    pool_path = Path(__file__).resolve().parent.parent / "data" / "vless_pool.json"
    try:
        with pool_path.open("r", encoding="utf-8") as fh:
            data = json.load(fh)
    except (OSError, json.JSONDecodeError):
        return None
    hosts: set[str] = set()
    for node in data.get("nodes", []) or []:
        host = node.get("host")
        if isinstance(host, str) and host:
            hosts.add(host)
    return hosts
```

## Tests — new `tests/test_xray_drift_health.py` (7 tests)

Unit tests against `_compute_xray_drift_block` with mocked `_extract_running_xray_hosts_for_health` and `_load_admitted_host_set`.

```python
"""v1.21 Phase 69 — xray drift block in /api/health/deep."""
from __future__ import annotations

import time as _time
import pytest
import backend.main as bm


@pytest.fixture(autouse=True)
def _reset_drift_state():
    bm._DRIFT_FIRST_SEEN.clear()
    yield
    bm._DRIFT_FIRST_SEEN.clear()


def test_xray_drift_block_absent_when_pool_unavailable():
    block, reason, critical = bm._compute_xray_drift_block(
        {"available": False}, cycle_age=None
    )
    assert block is None
    assert reason is None
    assert critical is False


def test_xray_drift_block_absent_when_config_unreadable(monkeypatch):
    monkeypatch.setattr(bm, "_extract_running_xray_hosts_for_health", lambda: None)
    block, reason, critical = bm._compute_xray_drift_block(
        {"available": True, "size": 3}, cycle_age=60
    )
    assert block is None
    assert reason is None
    assert critical is False


def test_xray_drift_block_zero_when_sets_match(monkeypatch):
    monkeypatch.setattr(
        bm, "_extract_running_xray_hosts_for_health", lambda: {"a", "b", "c"}
    )
    monkeypatch.setattr(bm, "_load_admitted_host_set", lambda: {"a", "b", "c"})
    block, reason, critical = bm._compute_xray_drift_block(
        {"available": True, "size": 3}, cycle_age=60
    )
    assert block == {
        "admitted_hosts": 3,
        "active_outbounds": 3,
        "drift_count": 0,
        "drifted_hosts": [],
        "first_seen_at": None,
    }
    assert reason is None
    assert critical is False


def test_xray_drift_block_reports_drift_count(monkeypatch):
    monkeypatch.setattr(
        bm, "_extract_running_xray_hosts_for_health", lambda: {"a", "b"}
    )
    monkeypatch.setattr(bm, "_load_admitted_host_set", lambda: {"b", "c", "d"})
    block, reason, critical = bm._compute_xray_drift_block(
        {"available": True, "size": 3}, cycle_age=60
    )
    # sym diff = {"a","c","d"}
    assert block["drift_count"] == 3
    assert block["drifted_hosts"] == ["a", "c", "d"]
    assert block["first_seen_at"] is not None
    # Just-detected drift within grace → no reason yet
    assert reason is None
    assert critical is False


def test_xray_drift_reason_added_after_5_min(monkeypatch):
    monkeypatch.setattr(
        bm, "_extract_running_xray_hosts_for_health", lambda: {"a"}
    )
    monkeypatch.setattr(bm, "_load_admitted_host_set", lambda: {"b"})
    # Seed _DRIFT_FIRST_SEEN with a stale first-seen timestamp to simulate
    # drift persisted > 5 min (avoid sleeping in tests).
    now = _time.monotonic()
    bm._DRIFT_FIRST_SEEN[frozenset({"a", "b"})] = (now - 301, "2026-05-12T19:00:00")
    block, reason, critical = bm._compute_xray_drift_block(
        {"available": True, "size": 1}, cycle_age=60
    )
    assert block["drift_count"] == 2
    assert reason == "xray_stale_config:2_nodes_drifted"
    assert critical is False  # cycle_age still fresh


def test_xray_drift_unhealthy_when_also_stale_cycle(monkeypatch):
    monkeypatch.setattr(
        bm, "_extract_running_xray_hosts_for_health", lambda: {"a"}
    )
    monkeypatch.setattr(bm, "_load_admitted_host_set", lambda: {"b"})
    now = _time.monotonic()
    bm._DRIFT_FIRST_SEEN[frozenset({"a", "b"})] = (now - 310, "2026-05-12T19:00:00")
    # cycle_age > 600s (10 min) with drift → critical
    block, reason, critical = bm._compute_xray_drift_block(
        {"available": True, "size": 1}, cycle_age=700
    )
    assert reason == "xray_stale_config:2_nodes_drifted"
    assert critical is True


def test_xray_drift_first_seen_resets_on_set_change(monkeypatch):
    monkeypatch.setattr(
        bm, "_extract_running_xray_hosts_for_health", lambda: {"a"}
    )
    monkeypatch.setattr(bm, "_load_admitted_host_set", lambda: {"b"})
    # First observation of drift {"a", "b"}
    bm._compute_xray_drift_block(
        {"available": True, "size": 1}, cycle_age=60
    )
    assert frozenset({"a", "b"}) in bm._DRIFT_FIRST_SEEN

    # Change admitted set → new drift key → old entry pruned.
    monkeypatch.setattr(bm, "_load_admitted_host_set", lambda: {"c"})
    bm._compute_xray_drift_block(
        {"available": True, "size": 1}, cycle_age=60
    )
    assert frozenset({"a", "b"}) not in bm._DRIFT_FIRST_SEEN
    assert frozenset({"a", "c"}) in bm._DRIFT_FIRST_SEEN
```

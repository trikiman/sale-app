# 68-02 — Wire reload helper into `refresh_proxy_list` + extend `pool_refresh_complete` event

**Phase:** 68 xray Auto-Reload on Admission Change
**Requirements:** REL-14, OBS-07 (partial — event schema extension)
**Depends on:** 68-01 (helpers must exist)

## Goal

Call `_reload_xray_systemd()` from `refresh_proxy_list` after a successful pool rewrite when the admitted host set differs from the running-config set. Extend the existing event emission to include restart outcome, duration, added/removed hosts, and before/after admitted host lists. Unchanged admission set MUST NOT trigger a restart.

## Change

### `vless/manager.py::refresh_proxy_list` — add admission-diff + reload call

Locate the tail of `refresh_proxy_list` after the admission logic writes the pool but before the `for node in admitted: self._track_event("vless_node_admitted", ...)` loop. Current tail (abridged):

```python
        new_pool = pool_state.replace_nodes(self._pool, admitted, verified_country="RU")
        with self._lock:
            self._pool = new_pool
            pool_state.save(self._pool, self._pool_path)
            self._rebuild_and_restart_xray()
        for node in admitted:
            self._track_event(
                "vless_node_admitted",
                {"host": node.host, "port": node.port, "name": node.name[:120]},
            )
        return len(admitted)
```

Replace with:

```python
        running_before = self._extract_running_hosts()
        new_pool = pool_state.replace_nodes(self._pool, admitted, verified_country="RU")
        admitted_hosts_after = {n.host for n in admitted}
        added_hosts = sorted(admitted_hosts_after - running_before)
        removed_hosts = sorted(running_before - admitted_hosts_after)
        with self._lock:
            self._pool = new_pool
            pool_state.save(self._pool, self._pool_path)
            self._rebuild_and_restart_xray()

        # REL-14: reload xray if the admitted host set differs from what
        # the running config holds. Unchanged set → "unchanged" outcome.
        if admitted_hosts_after == running_before:
            restart_outcome: str = "unchanged"
            restart_duration_ms: int | None = None
            restart_stderr_tail: str | None = None
        else:
            restart_outcome, restart_duration_ms, restart_stderr_tail = self._reload_xray_systemd()

        # OBS-07: emit pool_refresh_complete with full admission diff and
        # restart outcome. Existing admission events below stay for
        # back-compat with admin tools that read per-node entries.
        self._track_event(
            "pool_refresh_complete",
            {
                "admitted_count": len(admitted),
                "admitted_hosts_before": sorted(running_before),
                "admitted_hosts_after": sorted(admitted_hosts_after),
                "added_hosts": added_hosts,
                "removed_hosts": removed_hosts,
                "xray_restart_triggered": restart_outcome == "ok",
                "restart_duration_ms": restart_duration_ms,
                "restart_outcome": restart_outcome,
                "restart_stderr_tail": restart_stderr_tail,
            },
        )
        if restart_outcome == "failed":
            # Separate "failed" event so operators can alert on the
            # narrower event type independently of every refresh.
            self._track_event(
                "xray_restart_failed",
                {
                    "duration_ms": restart_duration_ms,
                    "stderr_tail": restart_stderr_tail,
                    "added_hosts": added_hosts,
                    "removed_hosts": removed_hosts,
                },
            )

        for node in admitted:
            self._track_event(
                "vless_node_admitted",
                {"host": node.host, "port": node.port, "name": node.name[:120]},
            )
        return len(admitted)
```

```python
def test_refresh_skips_reload_when_admitted_set_unchanged(pm, monkeypatch):
    """Pool refresh with identical host set must NOT trigger systemctl."""
    # Seed active.json with same host set as admitted will end up being.
    pm._xray_config_path.write_text(json.dumps({
        "outbounds": [
            {"tag": "vless-1", "protocol": "vless",
             "settings": {"vnext": [{"address": "1.2.3.4"}]}},
        ]
    }))
    call_log = []
    monkeypatch.setattr(
        pm, "_reload_xray_systemd",
        lambda: call_log.append("called") or ("ok", 10, None),
    )
    # Synthesize a minimal admission round: patch the candidate pipeline
    # to return a fresh node with host=1.2.3.4.
    from vless.parser import VlessNode
    fake = VlessNode(
        host="1.2.3.4", port=443, uuid="u", name="n",
        flow="", security="", encryption="none", type_="tcp",
        sni="", pbk="", sid="", spx="", fp="", extra={},
    )
    monkeypatch.setattr(pm, "_probe_candidates_in_parallel", lambda cands: [fake])
    monkeypatch.setattr(
        "vless.manager.sources.fetch_igareck_list", lambda: "irrelevant"
    )
    monkeypatch.setattr(
        "vless.manager.sources.parse_vless_list", lambda t: ([fake], [])
    )
    monkeypatch.setattr(
        "vless.manager.sources.filter_ru_nodes", lambda ns: ([fake], [])
    )
    pm.refresh_proxy_list()
    # _reload_xray_systemd should NOT have been called — set unchanged.
    assert call_log == []


def test_refresh_triggers_reload_when_admitted_set_differs(pm, monkeypatch):
    """Pool refresh with different host set MUST call _reload_xray_systemd."""
    # active.json has 1.1.1.1; admission will yield 9.9.9.9 — diff forces reload.
    from vless.parser import VlessNode
    fake = VlessNode(
        host="9.9.9.9", port=443, uuid="u", name="n",
        flow="", security="", encryption="none", type_="tcp",
        sni="", pbk="", sid="", spx="", fp="", extra={},
    )
    call_log: list[str] = []
    monkeypatch.setattr(
        pm, "_reload_xray_systemd",
        lambda: call_log.append("called") or ("ok", 42, None),
    )
    monkeypatch.setattr(pm, "_probe_candidates_in_parallel", lambda cands: [fake])
    monkeypatch.setattr(
        "vless.manager.sources.fetch_igareck_list", lambda: "irrelevant"
    )
    monkeypatch.setattr(
        "vless.manager.sources.parse_vless_list", lambda t: ([fake], [])
    )
    monkeypatch.setattr(
        "vless.manager.sources.filter_ru_nodes", lambda ns: ([fake], [])
    )
    pm.refresh_proxy_list()
    assert call_log == ["called"]
    # pool_refresh_complete event must be in events.jsonl with
    # restart_outcome=ok, admitted_hosts_after=[9.9.9.9], added_hosts=[9.9.9.9].
    lines = [json.loads(line) for line in pm._events_path.read_text().splitlines() if line.strip()]
    completes = [e for e in lines if e.get("event") == "pool_refresh_complete"]
    assert len(completes) == 1
    evt = completes[0]
    assert evt["restart_outcome"] == "ok"
    assert evt["xray_restart_triggered"] is True
    assert evt["admitted_hosts_after"] == ["9.9.9.9"]
    assert evt["added_hosts"] == ["9.9.9.9"]
    assert evt["removed_hosts"] == ["1.1.1.1", "2.2.2.2"]
    assert evt["restart_duration_ms"] == 42
```

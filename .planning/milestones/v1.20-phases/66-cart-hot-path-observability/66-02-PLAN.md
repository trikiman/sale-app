## Goal

Emit one JSONL line per cart-add attempt resolution (success, failed, auth_expired, product_gone, timeout, transient, pending_timeout, dedupe_hit, exception) to data/cart_events.jsonl for offline post-mortem. Every line carries the 11-key schema locked in 66-CONTEXT.md.

## Contract

### Emit point
Single helper _emit_cart_event(...) called from every terminal branch of cart_add_endpoint. Writes best-effort (disk failure MUST NOT block the HTTP response).

### Schema (exactly 11 keys per line)

- timestamp_iso — str, ISO-8601 UTC
- user_id_hash — str, sha256(user_id)[:12] via keepalive.warmup.hash_user_id
- attempt_id — str or null
- product_id — int
- duration_ms — int
- success — bool
- error_type — str or null
- client_request_id — str or null
- sessid_age_s — int or null (from cookies.json sessid_ts; null if file/field missing)
- warmup_hit — bool (true if keepalive.warmup._LAST_WARMUP_AT[user_id_hash] within 300 s)
- concurrent_recalc — bool (true if another pending attempt for same user_id with started_at within 12 s)

### Non-blocking guarantee
- All 3 detectors swallow exceptions and return safe defaults.
- _emit_cart_event swallows disk errors.
- Snapshot of warmup_hit + concurrent_recalc happens ONCE at the top of cart_add_endpoint (after _validate_user_header).

### New helpers (module-level, before cart_add_endpoint)

```python
def _read_sessid_age_seconds(user_id: str) -> int | None:
    """Peek sessid_ts from cookies.json without touching VkusVillCart."""
    try:
        cookies_path = _resolve_cart_cookies_path(user_id)
        with open(cookies_path, "r", encoding="utf-8") as f:
            data = json.load(f)
        if isinstance(data, dict):
            ts = data.get("sessid_ts")
            if ts:
                return max(0, int(_time.time() - float(ts)))
    except Exception:
        pass
    return None


def _warmup_hit_for_user(user_id: str) -> bool:
    try:
        from keepalive import warmup as _kw
        uid_hash = _kw.hash_user_id(str(user_id))
        with _kw._STATE_LOCK:
            last = _kw._LAST_WARMUP_AT.get(uid_hash)
        if last is None:
            return False
        return (_time.monotonic() - last) <= _CART_EVENT_WARMUP_FRESHNESS_S
    except Exception:
        return False


def _has_concurrent_recalc(user_id: str) -> bool:
    try:
        now = _time.time()
        cutoff = now - _CART_EVENT_CONCURRENT_RECALC_WINDOW_S
        with _cart_add_attempts_lock:
            for att in _cart_add_attempts.values():
                if (
                    str(att.get("user_id")) == str(user_id)
                    and att.get("status") == "pending"
                    and att.get("started_at", 0) >= cutoff
                ):
                    return True
    except Exception:
        pass
    return False


def _emit_cart_event(
    user_id: str,
    attempt_id: str | None,
    product_id: int,
    duration_ms: int,
    success: bool,
    error_type: str | None,
    client_request_id: str | None,
    sessid_age_s: int | None,
    warmup_hit: bool,
    concurrent_recalc: bool,
) -> None:
    try:
        from keepalive.warmup import hash_user_id
        line = json.dumps(
            {
                "timestamp_iso": datetime.now(timezone.utc).isoformat(),
                "user_id_hash": hash_user_id(str(user_id)),
                "attempt_id": attempt_id,
                "product_id": int(product_id),
                "duration_ms": int(duration_ms),
                "success": bool(success),
                "error_type": error_type,
                "client_request_id": client_request_id,
                "sessid_age_s": sessid_age_s,
                "warmup_hit": bool(warmup_hit),
                "concurrent_recalc": bool(concurrent_recalc),
            },
            ensure_ascii=False,
        )
        os.makedirs(os.path.dirname(_CART_EVENTS_PATH), exist_ok=True)
        with open(_CART_EVENTS_PATH, "a", encoding="utf-8") as f:
            f.write(line + "\n")
    except Exception:
        logger.debug("cart_events emit failed", exc_info=False)
```

Shape (matching `scripts/verify_v1.22.sh`):

```bash
#!/usr/bin/env bash
# v1.23 Detail-Path Performance + UX Polish smoke script.
# Grows phase-by-phase. Chains verify_v1.22.sh all at the end so v1.19-v1.22
# smoke gates stay green.
set -euo pipefail
cd "$(dirname "$0")/.."

MODE="${1:-phase74}"

run_phase74() {
    echo "=== v1.23 Phase 74: Product Details Cold-Path Latency ==="

    # PERF-10: cold-path p95 <= 2s across 5 never-cached product IDs.
    local base_url="${VERIFY_BASE_URL:-http://127.0.0.1:8000}"
    local ids=(100069 33215 40123 55666 77888)
    local times=()
    for pid in "${ids[@]}"; do
        # Force cache miss by appending ?_nocache=<random> if supported;
        # else rely on never-opened product_ids provided via $VERIFY_COLD_IDS.
        local t
        t=$(curl -s -o /dev/null -w "%{time_total}" \
            "${base_url}/api/product/${pid}/details")
        times+=("$t")
        echo "  product ${pid}: ${t}s"
    done

    # PERF-11: verify data/detail_events.jsonl has entries for this run.
    local ledger="data/detail_events.jsonl"
    if [[ ! -f "$ledger" ]]; then
        echo "FAIL: ${ledger} missing"
        exit 1
    fi
    local ledger_lines
    ledger_lines=$(wc -l < "$ledger")
    if [[ "$ledger_lines" -lt 5 ]]; then
        echo "FAIL: ${ledger} has ${ledger_lines} lines, expected >= 5"
        exit 1
    fi
    echo "  ledger OK (${ledger_lines} lines)"

    # Ledger schema spot-check: every line has the 6 required keys
    python3 - <<'PYEOF'
import json, sys
required = {"ts","product_id","duration_ms","cached","retry_count","outcome"}
with open("data/detail_events.jsonl") as fh:
    for i, line in enumerate(fh, 1):
        e = json.loads(line)
        missing = required - set(e.keys())
        if missing:
            sys.exit(f"FAIL: line {i} missing keys {missing}")
print("  schema OK")
PYEOF

    echo "=== v1.23 Phase 74 smoke: GREEN ==="
}

case "$MODE" in
    phase74) run_phase74 ;;
    all)
        run_phase74
        echo ""
        bash scripts/verify_v1.22.sh all
        ;;
    *) echo "usage: $0 [phase74|all]"; exit 2 ;;
esac
```
